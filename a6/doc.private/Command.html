Module Command

module Command: sig .. end
type entry_phrase = string list 
type command =
| Fire of entry_phrase
| Quit
| Status
| Place of entry_phrase 
exception Empty
exception Malformed
val shave : entry_phrase -> entry_phrase -> entry_phrase

shave entry acc is entry with all "" elements removed. These elements appear as a result of the String.split_on_char function.

val parseHelper : entry_phrase -> command

parseHelper entry is the command corresponding to the given entry phrase.

val parse : string -> command

parse is the command corresponding to the given string of user input.
